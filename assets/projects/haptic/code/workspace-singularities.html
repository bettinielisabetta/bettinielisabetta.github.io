Workspace & Singularities Plot

MATLAB implementation used to compute the five-bar linkage workspace and identify singular configurations. 
clear; close all; clc

la = 140;
lb = 180;
lc = 120;

C = [];
W_L = [];
W_P = [];
W_W = [];
W_S = [];
Singul = [];

for teta1 = (0):(pi/100):(2*pi)
    for teta4 = (0):(pi/100):(2*pi)

        sol_W = zeros(2,2);
        sol_S = zeros(8,1);
        singul = zeros(6,1);

        [sol_C, sol_L, sol_P, sol_W, sol_S, singul] = ...
            endeffectorpositionsingnew([la, lb, lc, teta1, teta4]);

        W_P = [W_P, sol_P];
        W_W = [W_W, sol_W];
        W_S = [W_S, sol_S];

    end
end

function [out_conf, out_L, out_P, out_W, out_S, singul] = endeffectorpositionsingnew(in)

    la = in(1);
    lb = in(2);
    lc = in(3);
    teta1 = in(4);
    teta4 = in(5);

    c4 = cos(teta4);
    s4 = sin(teta4);
    c1 = cos(teta1);
    s1 = sin(teta1);

    singul = [false; false; false; false; false; false];

    J_1 = zeros(2,2);
    J_2 = zeros(2,2);

    a_1 = [0; 0; 0; 0; 0];
    a_2 = [0; 0; 0; 0; 0];
    b_1 = [0; 0; 0; 0; 0];
    b_2 = [0; 0; 0; 0; 0];

    % End-effector coordinate calculation
    E = 2 * lb * (lc + la * (c4 - c1));
    F = 2 * la * lb * (s4 - s1);
    G = lc^2 + 2 * la^2 + 2 * lc * la * c4 - ...
        2 * lc * la * c1 - 2 * (la^2) * cos(teta4 - teta1);

    RADW = E^2 + F^2 - G^2;

    if RADW >= 0

        SQRT_RADW = sqrt(RADW);
        SQRT_RADW_P = (-F + SQRT_RADW) / (G - E);
        SQRT_RADW_N = (-F - SQRT_RADW) / (G - E);

        xp1 = lc + la * c4 + lb * cos(2 * atan(SQRT_RADW_P));
        yp1 = la * s4 + lb * sin(2 * atan(SQRT_RADW_P));

        xp2 = lc + la * c4 + lb * cos(2 * atan(SQRT_RADW_N));
        yp2 = la * s4 + lb * sin(2 * atan(SQRT_RADW_N));

        % Link endpoint coordinates for plotting
        a_1 = [0; la * c1; xp1; lc + la * c4; lc];
        a_2 = [0; la * c1; xp2; lc + la * c4; lc];

        b_1 = [0; la * s1; yp1; la * s4; 0];
        b_2 = [0; la * s1; yp2; la * s4; 0];

        % Theta2 and theta3 calculation
        teta2_1 = angleCalc((yp1 - la * s1) / lb, (xp1 - la * c1) / lb, 'rad');
        teta2_2 = angleCalc((yp2 - la * s1) / lb, (xp2 - la * c1) / lb, 'rad');

        teta3_1 = angleCalc((yp1 - la * s4) / lb, (xp1 - lc - la * c4) / lb, 'rad');
        teta3_2 = angleCalc((yp2 - la * s4) / lb, (xp2 - lc - la * c4) / lb, 'rad');

        % Singularity calculation
        eps = 0.1;

        singul1_1 = (abs(sin(teta2_1 - teta1) * sin(teta3_1 - teta4)) < eps);
        singul2_1 = (abs(sin(teta3_1 - teta2_1)) < eps);
        singul3_1 = (abs(sin(teta2_1 - teta1) * sin(teta3_1 - teta4)) < eps) & ...
                    (abs(sin(teta3_1 - teta2_1)) < eps);

        singul1_2 = (abs(sin(teta2_2 - teta1) * sin(teta3_2 - teta4)) < eps);
        singul2_2 = (abs(sin(teta3_2 - teta2_2)) < eps);
        singul3_2 = (abs(sin(teta2_2 - teta1) * sin(teta3_2 - teta4)) < eps) & ...
                    (abs(sin(teta3_2 - teta2_2)) < eps);

        singul = [
            singul1_1;
            singul2_1;
            singul3_1;
            singul1_2;
            singul2_2;
            singul3_2
        ];

        if singul1_1 == 1
            a = 1;
        elseif singul2_1 == 1
            a = 2;
        elseif singul3_1 == 1
            a = 3;
        else
            a = 0;
        end

        if singul1_2 == 1
            b = 1;
        elseif singul2_2 == 1
            b = 2;
        elseif singul3_2 == 1
            b = 3;
        else
            b = 0;
        end

        % Jacobian calculation
        % Aj_1 = [lb*cos(teta2_1) lb*sin(teta2_1);
        %         lb*cos(teta3_1) lb*sin(teta3_1)];
        %
        % Aj_2 = [lb*cos(teta2_2) lb*sin(teta2_2);
        %         lb*cos(teta3_2) lb*sin(teta3_2)];
        %
        % Bj_1 = [la*lb*sin(teta2_1-teta1) 0;
        %         0 la*lb*sin(teta3_1-teta4)];
        %
        % Bj_2 = [la*lb*sin(teta2_2-teta1) 0;
        %         0 la*lb*sin(teta3_2-teta4)];
        %
        % J_1 = -(Aj_1^-1) * Bj_1;
        % J_2 = -(Aj_2^-1) * Bj_2;

    else

        xp1 = NaN;
        yp1 = NaN;
        xp2 = NaN;
        yp2 = NaN;

        teta2_1 = NaN;
        teta2_2 = NaN;
        teta3_1 = NaN;
        teta3_2 = NaN;

        a = NaN;
        b = NaN;

    end

    out_W(1,1) = xp1;
    out_W(2,1) = yp1;
    out_W(1,2) = xp2;
    out_W(2,2) = yp2;

    out_P(1,1) = xp1;
    out_P(2,1) = yp1;
    out_P(3,1) = xp2;
    out_P(4,1) = yp2;

    out_L(1:5,1) = a_1;
    out_L(6:10,1) = b_1;
    out_L(11:15,1) = a_2;
    out_L(16:20,1) = b_2;

    out_S(1,1) = cond(J_1);
    out_S(2,1) = cond(J_2);
    out_S(3,1) = teta1;
    out_S(4,1) = teta4;
    out_S(5,1) = xp1;
    out_S(6,1) = yp1;
    out_S(7,1) = xp2;
    out_S(8,1) = yp2;

    out_conf(1,1) = teta1;
    out_conf(1,2) = teta4;
    out_conf(1,3) = teta2_1;
    out_conf(1,4) = teta3_1;
    out_conf(1,5) = xp1;
    out_conf(1,6) = yp1;
    out_conf(1,7) = teta2_2;
    out_conf(1,8) = teta3_2;
    out_conf(1,9) = xp2;
    out_conf(1,10) = yp2;
    out_conf(1,11) = la * c1;
    out_conf(1,12) = la * s1;
    out_conf(1,13) = lc + la * c4;
    out_conf(1,14) = la * s4;
    out_conf(1,15) = a;
    out_conf(1,16) = b;

end


function theta = angleCalc(S, C, out_mode)

    if nargin < 3
        out_mode = 'deg';
    end

    if strcmp(out_mode, 'deg')
        cons = 180 / pi;
    else
        cons = 1;
    end

    theta = asin(S);

    if C < 0
        if S > 0
            theta = pi - theta;
        elseif S < 0
            theta = -pi - theta;
        else
            theta = theta + pi;
        end
    end

    theta = theta * cons;

end
