MATLAB implementation used for kinetostatic joint torque evaluation and actuator sizing of the five-bar linkage.

for teta1 = (0):(pi/100):(2*pi)
    for teta4 = (0):(pi/100):(2*pi)

        sol_W = zeros(2,2);
        sol_S = zeros(8,1);
        singul = zeros(6,1);

        [sol_C, sol_L, sol_P, sol_W, sol_S, singul] = ...
            endeffectorpositionsingnew([la, lb, lc, teta1, teta4]);

        C = [C; sol_C];
        W_L = [W_L, sol_L];
        W_P = [W_P, sol_P];
        W_W = [W_W, sol_W];
        W_S = [W_S, sol_S];
        Singul = [Singul, singul];

        if (sol_W(1,1) > 0) && (sol_W(1,1) < 120) && ...
           (sol_W(2,1) > 120) && (sol_W(2,1) < 220)

            [sol_T] = tau([
                la, lb, lc, ...
                teta1, teta4, ...
                sol_C(1,3), sol_C(1,7), sol_C(1,4), sol_C(1,8)
            ]);

            w1 = [sol_W(1,1), sol_W(2,1), sol_T];
            W1 = [W1; w1];

        end

        % Second configuration

%% FUNCTION

function [out_T] = tau(in)

    la = in(1);
    lb = in(2);
    lc = in(3);
    teta1 = in(4);
    teta4 = in(5);
    teta2_1 = in(6);
    teta2_2 = in(7);
    teta3_1 = in(8);
    teta3_2 = in(9);

    c4 = cos(teta4);
    s4 = sin(teta4);
    c1 = cos(teta1);
    s1 = sin(teta1);

    s12_1 = sin(teta1 + teta2_1);
    s12_2 = sin(teta1 + teta2_2);
    c12_1 = cos(teta1 + teta2_1);
    c12_2 = cos(teta1 + teta2_2);

    s34_1 = sin(teta4 + teta3_1);
    s34_2 = sin(teta4 + teta3_2);
    c34_1 = cos(teta4 + teta3_1);
    c34_2 = cos(teta4 + teta3_2);

    T = [];
    T_tot = [];
    F_tot = [];

    alpha = pi/4;
    F = 35;

    teta21 = 2*pi - (teta1 - teta2_1);
    teta22 = 2*pi - (teta1 - teta2_2);

    teta31 = teta3_1 - teta4;
    teta32 = teta3_2 - teta4;

    s121 = sin(teta1 + teta21);
    s122 = sin(teta1 + teta22);
    c121 = cos(teta1 + teta21);
    c122 = cos(teta1 + teta22);

    s341 = sin(teta4 + teta31);
    s342 = sin(teta4 + teta32);
    c341 = cos(teta4 + teta31);
    c342 = cos(teta4 + teta32);

    fx1_1 = (cot(teta1 + teta21)) * F * ...
            ((cos(alpha) - sin(alpha) * (cot(teta31 + teta4))) / ...
            ((cot(teta31 + teta4)) - (cot(teta1 + teta21))));

    fy1_1 = F * ...
            ((cos(alpha) - sin(alpha) * (cot(teta31 + teta4))) / ...
            ((cot(teta31 + teta4)) - (cot(teta1 + teta21))));

    fx2_1 = (cot(teta31 + teta4)) * F * ...
            ((-cos(alpha) + sin(alpha) * (cot(teta1 + teta21))) / ...
            ((cot(teta31 + teta4)) - (cot(teta1 + teta21))));

    fy2_1 = F * ...
            ((-cos(alpha) + sin(alpha) * (cot(teta1 + teta21))) / ...
            ((cot(teta31 + teta4)) - (cot(teta1 + teta21))));

    tau1_1 = F * ...
             ((cos(alpha) - sin(alpha) * (cot(teta31 + teta4))) / ...
             ((cot(teta31 + teta4)) - (cot(teta1 + teta21)))) * ...
             (la * c1 + lb * c121 - ...
             (la * s1 + lb * s121) * (cot(teta1 + teta21)));

    tau4_1 = F * ...
             ((-cos(alpha) + sin(alpha) * (cot(teta1 + teta21))) / ...
             ((cot(teta31 + teta4)) - (cot(teta1 + teta21)))) * ...
             ((cot(teta31 + teta4)) * (-lb * s341 - la * s4) + ...
             lb * c341 + la * c4);

    T(1,1) = tau1_1;
    T(1,2) = tau4_1;

    out_T = T;

end
