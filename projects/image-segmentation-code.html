MATLAB Implementation

MATLAB script used for manuscript image preprocessing, foreground segmentation, connected component analysis, feature extraction, interactive labeling, and supervised text/non-text classification.

Code shown for portfolio review. The script is displayed as part of the project documentation. 
%% ============================================================
%  Segmentation and Supervised Classification in Historical Manuscripts
%  (Spices 17 - Bettini Elisabetta)
%
%  Pipeline summary:
%   1) Read + inspect input ROI
%   2) Grayscale + normalization
%   3) Contrast enhancement + denoising
%   4) Dual-threshold (Otsu) + morphological reconstruction
%   5) Morphological refinement (area opening, closing, limited hole filling)
%   6) Decoration splitting (area-based separation + marker-controlled watershed)
%   7) Connected components + feature extraction
%   8) Interactive labeling + supervised classification (Linear SVM)
%   9) Post-processing + visualization
%% ============================================================

clear; close all; clc;

%% 1. Input image (ROI) and basic inspection
imgPath = 'my_roi.jpg';
Irgb = imread(imgPath);

disp('--- Image inspection ---');
disp(['File:  ', imgPath]);
disp(['Class: ', class(Irgb)]);
disp(['Size:  ', mat2str(size(Irgb))]);

figure; imshow(Irgb);
title('Figure 1: Original ROI (RGB)');

%% 2. Grayscale conversion and normalization
if size(Irgb,3) == 3
    Igray = rgb2gray(Irgb);
else
    Igray = Irgb;
end
Igray = im2double(Igray);

figure; imshow(Igray);
title('Figure 2: Grayscale ROI (normalized to [0,1])');

%% 3. Intensity analysis and contrast enhancement
figure; imhist(Igray);
title('Figure 3: Grayscale histogram');

Iadj = imadjust(Igray);

figure;
subplot(1,2,1); imshow(Igray); title('Before contrast adjustment');
subplot(1,2,2); imshow(Iadj);  title('After contrast adjustment');
sgtitle('Figure 4: Contrast stretching (imadjust)');

figure;
subplot(1,2,1); imhist(Igray); title('Histogram before');
subplot(1,2,2); imhist(Iadj);  title('Histogram after');
sgtitle('Figure 5: Histogram comparison');

%% 4. Noise reduction (Gaussian filtering)
sigma = 1.0;  % typical range: 0.8–1.5
Ifilt = imgaussfilt(Iadj, sigma);

figure;
subplot(1,2,1); imshow(Iadj);  title('After contrast adjustment');
subplot(1,2,2); imshow(Ifilt); title(['After Gaussian filtering (\sigma = ', num2str(sigma), ')']);
sgtitle('Figure 6: Denoising');

%% 5. Foreground segmentation: dual-threshold Otsu + reconstruction
Totsu = graythresh(Ifilt);

kSeed = 0.90;   % conservative: confident ink only
kCand = 1.15;   % permissive: includes lighter strokes (typical: 1.02–1.20)

BW_seed = Ifilt < (kSeed*Totsu);
BW_cand = Ifilt < (kCand*Totsu);

BW_rec = imreconstruct(BW_seed, BW_cand);

figure;
subplot(1,3,1); imshow(BW_seed); title(['Seed mask (k = ', num2str(kSeed), ')']);
subplot(1,3,2); imshow(BW_cand); title(['Candidate mask (k = ', num2str(kCand), ')']);
subplot(1,3,3); imshow(BW_rec);  title('Reconstructed foreground');
sgtitle('Figure 7: Dual-threshold + reconstruction');

%% 6. Morphological refinement (clean + reconnect + limited hole filling)
% 6.1 Remove small components
minArea = 150;  % tuned to ROI resolution/scale
BW_clean1 = bwareaopen(BW_rec, minArea);

figure;
subplot(1,2,1); imshow(BW_rec);    title('After reconstruction');
subplot(1,2,2); imshow(BW_clean1); title(['After area opening (minArea = ', num2str(minArea), ')']);
sgtitle('Figure 8: Small object removal');

% 6.2 Closing (reconnect fragmented strokes / leaf edges)
seClose = strel('disk', 1);
BW_clean2 = imclose(BW_clean1, seClose);

figure;
subplot(1,2,1); imshow(BW_clean1); title('After area opening');
subplot(1,2,2); imshow(BW_clean2); title('After closing (disk r = 1)');
sgtitle('Figure 9: Closing');

% 6.3 Limited hole filling (fill only small holes)
holes = imfill(BW_clean2, 'holes') & ~BW_clean2;
CC_h = bwconncomp(holes);
holeAreas = cellfun(@numel, CC_h.PixelIdxList);

maxHoleArea = 50;   % tuned 50–1500
keep = find(holeAreas <= maxHoleArea);

holesSmall = false(size(BW_clean2));
for ii = keep
    holesSmall(CC_h.PixelIdxList{ii}) = true;
end

BW_clean3 = BW_clean2 | holesSmall;

figure;
subplot(1,3,1); imshow(BW_clean2);  title('After closing');
subplot(1,3,2); imshow(holesSmall); title(['Filled holes (Area \le ', num2str(maxHoleArea), ')']);
subplot(1,3,3); imshow(BW_clean3);  title('After limited hole filling');
sgtitle('Figure 10: Limited hole filling');

%% 7. Decoration splitting prior to connected components extraction
% Goal: improve separation of fused decorative structures while preserving text integrity.

minAreaCC = 250;  % minimum CC size 
BW_cc0 = bwareaopen(BW_clean3, minAreaCC);

figure; imshow(BW_cc0);
title('Figure 11: Base mask for connected components (BW\_cc0)');

% 7.1 Separate large vs small components (area-based)
CC0 = bwconncomp(BW_cc0);
stats0 = regionprops(CC0, 'Area');
A0 = [stats0.Area];

areaSplitThr = 6000;  % tuned: separates macro decorations/illustrations vs text-like CCs
idxBig = find(A0 >= areaSplitThr);

L0 = labelmatrix(CC0);
BW_big   = ismember(L0, idxBig);
BW_small = BW_cc0 & ~BW_big;

figure;
subplot(1,3,1); imshow(BW_cc0);   title('BW\_cc0');
subplot(1,3,2); imshow(BW_big);   title(['Large CCs (Area \ge ', num2str(areaSplitThr), ')']);
subplot(1,3,3); imshow(BW_small); title('Small CCs (kept intact)');
sgtitle('Figure 12: Large vs small component separation');

% 7.2 Split fused large components via marker-controlled watershed
BW = BW_big;

D = bwdist(~BW);
D = imgaussfilt(D, 1);  % tuned: 0.5–2

hMax = 1.5;             % tuned typical range: 0.8–3 
M = imextendedmax(D, hMax);
M = bwareaopen(M, 20);

D2 = imimposemin(-D, M | ~BW);
Lw = watershed(D2);

BW_ws = BW;
BW_ws(Lw == 0) = 0;
BW_ws = bwareaopen(BW_ws, minAreaCC);

figure;
subplot(2,3,1); imshow(BW);              title('Large CC mask (input)');
subplot(2,3,2); imagesc(D); axis image off; title('Distance transform'); colormap jet; colorbar;
subplot(2,3,3); imshow(M);               title(['Markers (h = ', num2str(hMax), ')']);
subplot(2,3,4); imagesc(Lw); axis image off; title('Watershed labels'); colormap jet; colorbar;
subplot(2,3,5); imshow(Lw==0);           title('Watershed ridges');
subplot(2,3,6); imshow(BW_ws);           title('Large CCs after watershed split');
sgtitle('Figure 13: Watershed splitting of fused decorations');

% 7.3 Recombine (keep text CCs intact)
BW_split = BW_ws | BW_small;
BW_cc = bwareaopen(BW_split, minAreaCC);

figure;
subplot(1,2,1); imshow(BW_cc0); title('Before split (BW\_cc0)');
subplot(1,2,2); imshow(BW_cc);  title('After split (BW\_cc)');
sgtitle('Figure 14: Final CC extraction mask');

% Feature measurement mask (less altered topology than BW_cc)
BW_feat = bwareaopen(BW_clean2, minAreaCC);

%% 8. Connected components and feature extraction
BW_cc   = bwareaopen(BW_cc,   minAreaCC);
BW_feat = bwareaopen(BW_feat, minAreaCC);

CC = bwconncomp(BW_cc);
disp(['Connected components (after split): ', num2str(CC.NumObjects)]);

Lcc = labelmatrix(CC);
figure; imagesc(Lcc); axis image off;
title('Figure 15: Labeled connected components');
colormap jet; colorbar;

props = regionprops(CC, BW_feat, ...
    'Area','Perimeter','Eccentricity','Solidity', ...
    'MajorAxisLength','MinorAxisLength', ...
    'Extent','EulerNumber', ...
    'Centroid','BoundingBox');

T = struct2table(props);

propsI = regionprops(CC, Ifilt, 'MeanIntensity','MinIntensity','MaxIntensity');
T = [T struct2table(propsI)];

% Derived features
eps0 = 1e-6;
T.AspectRatio = T.MajorAxisLength ./ (T.MinorAxisLength + eps0);
T.Circularity = 4*pi*T.Area ./ (T.Perimeter.^2 + eps0);

T.BB_Width  = T.BoundingBox(:,3);
T.BB_Height = T.BoundingBox(:,4);
T.BB_Area   = T.BB_Width .* T.BB_Height;
T.FillRatio = T.Area ./ (T.BB_Area + eps0);

T.StrokeProxy = T.Area ./ (T.Perimeter + eps0);

disp('--- Feature table (first rows) ---');
disp(T(1:min(10,height(T)), :));

% Bounding box visualization
figure; imshow(Irgb); hold on;
statsBB = regionprops(CC, 'BoundingBox');
for i = 1:numel(statsBB)
    rectangle('Position', statsBB(i).BoundingBox, 'EdgeColor','r','LineWidth',0.5);
end
title('Figure 16: Bounding boxes of connected components');
hold off;

%% 9. Feature matrix construction
featNames = {'Area','Perimeter','Eccentricity','Solidity', ...
             'MajorAxisLength','MinorAxisLength', ...
             'Extent','EulerNumber', ...
             'MeanIntensity','MinIntensity','MaxIntensity', ...
             'AspectRatio','Circularity','FillRatio','StrokeProxy'};

featNames = featNames(ismember(featNames, T.Properties.VariableNames));

X_all = T{:, featNames};
X_all_z = zscore(X_all);

disp(['Features used: ', num2str(size(X_all_z,2)), ' (standardized)']);

%% 10. Interactive labeling (Text vs NonText)
L = labelmatrix(CC);
propsL = regionprops(CC, Ifilt, 'BoundingBox', 'Centroid', 'Area');

figure; imshow(Iadj);
title('Interactive labeling: click a component (press Enter to stop)'); hold on;

labelNames = {'Text','NonText'};
labelCodes = [1 2];

idxTrain = [];
yTrain = [];

minAreaLabel = 80;
maxAreaLabel = 0.20 * numel(BW_cc);

while true
    [x, y] = ginput(1);
    if isempty(x); break; end

    r = round(y); c = round(x);
    if r < 1 || r > size(L,1) || c < 1 || c > size(L,2)
        disp('Click inside the image.');
        continue;
    end

    id = L(r, c);
    if id == 0
        disp('Background clicked. Try again.');
        continue;
    end

    A = propsL(id).Area;
    if A < minAreaLabel
        disp(['Skipping CC ', num2str(id), ' (too small: Area=', num2str(A), ').']);
        continue;
    end
    if A > maxAreaLabel
        disp(['Skipping CC ', num2str(id), ' (too large: Area=', num2str(A), ').']);
        continue;
    end

    bb = propsL(id).BoundingBox;
    rectangle('Position', bb, 'EdgeColor','y', 'LineWidth', 1.2);

    choice = menu(['Component ', num2str(id), ': choose class'], labelNames{:});
    if choice == 0
        disp('No class selected. Skipping.');
        continue;
    end

    idxTrain(end+1,1) = id;
    yTrain(end+1,1)   = labelCodes(choice);

    disp(['Labeled CC ', num2str(id), ' as ', labelNames{choice}, ' (Area=', num2str(A), ')']);
end
hold off;

[idxTrain, ia] = unique(idxTrain, 'stable');
yTrain = yTrain(ia);

disp(['Total labeled components: ', num2str(numel(idxTrain))]);

%% 11. Supervised classification (Linear SVM) 
Xtrain = X_all_z(idxTrain, :);
ytrain = yTrain;

disp('Training set size per class:');
disp(tabulate(ytrain));

Mdl = fitcsvm(Xtrain, ytrain, ...
    'KernelFunction','linear', ...
    'Standardize',false, ...
    'ClassNames',[1 2]);

K = min(5, numel(ytrain));
CV = crossval(Mdl, 'KFold', K);
loss = kfoldLoss(CV);
disp(['K-fold loss: ', num2str(loss), ' (accuracy ~ ', num2str(1-loss), ')']);

yPredAll = predict(Mdl, X_all_z);

% Post-classification cleanup (large CCs are unlikely to be text)
maxTextArea = 2000;  % tuned for ROI scale
yPredAll_clean = yPredAll;
yPredAll_clean((yPredAll == 1) & (T.Area > maxTextArea)) = 2;


%% 12. Visualization of predicted classes
propsAll = regionprops(CC, 'BoundingBox', 'Centroid', 'Area');

drawMinArea = 200;

figure; imshow(Iadj); hold on;
title('Figure 18: Predicted classes (Text = green, NonText = red)');
for i = 1:CC.NumObjects
    if propsAll(i).Area < drawMinArea
        continue;
    end
    bb = propsAll(i).BoundingBox;
    if yPredAll_clean(i) == 1
        rectangle('Position', bb, 'EdgeColor', 'g', 'LineWidth', 0.8);
    else
        rectangle('Position', bb, 'EdgeColor', 'r', 'LineWidth', 0.8);
    end
end
hold off;

figure;
subplot(1,2,1); imshow(Iadj); title('Figure 19a: Predicted TEXT'); hold on;
for i = 1:CC.NumObjects
    if yPredAll_clean(i) == 1 && propsAll(i).Area >= drawMinArea
        rectangle('Position', propsAll(i).BoundingBox, 'EdgeColor', 'g', 'LineWidth', 1);
    end
end
hold off;

subplot(1,2,2); imshow(Iadj); title('Figure 19b: Predicted NON-TEXT'); hold on;
for i = 1:CC.NumObjects
    if yPredAll_clean(i) == 2 && propsAll(i).Area >= drawMinArea
        rectangle('Position', propsAll(i).BoundingBox, 'EdgeColor', 'r', 'LineWidth', 1);
    end
end
hold off;
